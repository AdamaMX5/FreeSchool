# certbot installieren
sudo apt update

# Zertifikat erzeugen # Port 80 muss frei sein.

# Cronjob zur Erneuerung # gültig 3 Monate
sudo crontab -e
# Füge hinzu:
0 3 * * * certbot renew --quiet && systemctl reload nginx
